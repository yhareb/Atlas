            status TEXT NOT NULL DEFAULT 'OPEN',
            quantity REAL NOT NULL,

            exit_price REAL,

            entry_fees REAL DEFAULT 0,
            exit_fees REAL DEFAULT 0,
            realized_pnl REAL,
            realized_pnl_pct REAL,
            parent_id INTEGER,
            notes TEXT,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
        , stop_loss REAL, risk_pct REAL, target_price REAL, broker_ref TEXT DEFAULT NULL, manual_stop_lock INTEGER DEFAULT 0, current_price REAL, last_price REAL, last_price_at DATETIME);
CREATE INDEX idx_trades_ticker_status ON trades(ticker, status);
